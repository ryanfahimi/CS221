#include <stdio.h>

int main()
{
    int n, factorial = 1;
    printf("Enter a number: ");
    scanf("%d", &n);

    for (int i = 1; i <= n; i++)
    {
        factorial *= i;
    }

    printf("The Factorial of %d is: %d\n", n, factorial);
    return 0;
}

#include <stdio.h>

int main()
{
    long int num, sum = 0;
    printf("Enter a number: ");
    scanf("%ld", &num);

    if (num < 0)
    {
        num = -num;
    }

    while (num > 0)
    {
        sum += num % 10;
        num /= 10;
    }

    printf("The sum of the digits is: %ld\n", sum);
    return 0;
}

#include <stdio.h>

int gcd(int a, int b)
{
    if (b == 0)
    {
        return a;
    }
    return gcd(b, a % b);
}

int main()
{
    int a, b;
    printf("Enter the first number : ");
    scanf("%d", &a);
    printf("Enter the second number : ");
    scanf("%d", &b);

    int result = gcd(a, b);
    printf("The greatest common divisor of %d and %d is %d\n", a, b, result);
    return 0;
}
